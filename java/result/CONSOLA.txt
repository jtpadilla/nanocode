```java
import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.nodes.Element;
import org.jsoup.select.Elements;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;

/**
 * Clase UjiScraper para realizar scraping de noticias de la Universitat Jaume I.
 */
public class UjiScraper {

    /**
     * Record que representa una noticia.
     */
    public record Noticia(String titulo, String fecha, String enlace) {}

    /**
     * Realiza el scraping de la página de noticias de la UJI.
     *
     * @return Una lista de objetos Noticia.
     */
    public List<Noticia> scrapeNoticies() {
        List<Noticia> noticias = new ArrayList<>();
        String url = "https://www.uji.es/com/noticies/";

        try {
            // Conectar y obtener el documento HTML
            Document doc = Jsoup.connect(url).get();

            // Seleccionar los elementos de los títulos y las fechas
            // Basado en la estructura de la web de la UJI
            Elements elementosTitulo = doc.select(".titol-noticia");
            Elements elementosFecha = doc.select(".data-noticia");

            int numNoticias = Math.min(elementosTitulo.size(), elementosFecha.size());

            for (int i = 0; i < numNoticias; i++) {
                Element tituloElem = elementosTitulo.get(i);
                Element fechaElem = elementosFecha.get(i);

                String titulo = tituloElem.text();
                // El enlace se encuentra dentro de un tag 'a' en '.titol-noticia'
                String enlace = tituloElem.select("a").attr("abs:href");
                String fecha = fechaElem.text();

                noticias.add(new Noticia(titulo, fecha, enlace));
            }

        } catch (IOException e) {
            System.err.println("Error al acceder a la URL: " + e.getMessage());
        }

        return noticias;
    }

    /**
     * Método principal para ejecutar el scraper y mostrar los resultados.
     */
    public static void main(String[] args) {
        UjiScraper scraper = new UjiScraper();
        List<Noticia> listaNoticias = scraper.scrapeNoticies();

        // Mostrar las noticias obtenidas
        for (Noticia n : listaNoticias) {
            System.out.println(n);
        }
    }
}
```

import requests
from bs4 import BeautifulSoup

def generar_codigo_scraping_uji():
    url = "https://www.uji.es/com/noticies/"
    headers = {'User-Agent': 'UjiScraper-Bot'}

    response = requests.get(url, headers=headers)
    if response.status_code == 200:
        # Forzar codificación utf-8 para tildes y caracteres especiales
        response.encoding = 'utf-8'
        soup = BeautifulSoup(response.text, 'html.parser')

        titulos = soup.select(".titol-noticia")
        fechas = soup.select(".data-noticia")

        for t, f in zip(titulos, fechas):
            titulo = t.get_text(strip=True)
            # Obtener el enlace relativo y convertirlo a absoluto
            link_tag = t.find('a')
            enlace = link_tag['href'] if link_tag else ""
            if enlace and not enlace.startswith('http'):
                enlace = "https://www.uji.es" + enlace

            fecha = f.get_text(strip=True)

            print(f"Fecha: {fecha} | Título: {titulo}")
            print(f"Enlace: {enlace}\n")
    else:
        print(f"Error al acceder a la web: {response.status_code}")

if __name__ == "__main__":
    generar_codigo_scraping_uji()

### Código Python (Alternativa):

python
import requests
from bs4 import BeautifulSoup

def generar_codigo_scraping_uji():
    url = "https://www.uji.es/com/noticies/"
    headers = {'User-Agent': 'UjiScraper-Bot'}

    response = requests.get(url, headers=headers)
    if response.status_code == 200:
        # Forzar codificación utf-8 para tildes y caracteres especiales
        response.encoding = 'utf-8'
        soup = BeautifulSoup(response.text, 'html.parser')

        titulos = soup.select(".titol-noticia")
        fechas = soup.select(".data-noticia")

        for t, f in zip(titulos, fechas):
            titulo = t.get_text(strip=True)
            # Obtener el enlace relativo y convertirlo a absoluto
            link_tag = t.find('a')
            enlace = link_tag['href'] if link_tag else ""
            if enlace and not enlace.startswith('http'):
                enlace = "https://www.uji.es" + enlace

            fecha = f.get_text(strip=True)

            print(f"Fecha: {fecha} | Título: {titulo}")
            print(f"Enlace: {enlace}\n")
    else:
        print(f"Error al acceder a la web: {response.status_code}")

if __name__ == "__main__":
    generar_codigo_scraping_uji()

import org.jsoup.Jsoup;
import org.jsoup.nodes.Document;
import org.jsoup.nodes.Element;
import org.jsoup.select.Elements;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;
public class UjiScraper implements Scraper {
    @Override
    public List<Noticia> scrape() {
        List<Noticia> noticias = new ArrayList<>();
        String url = "https://www.uji.es/com/noticies/";
        try {
            Document doc = Jsoup.connect(url).get();
            Elements elementosTitulo = doc.select(".titol-noticia");
            Elements elementosFecha = doc.select(".data-noticia");
            int numNoticias = Math.min(elementosTitulo.size(), elementosFecha.size());
            for (int i = 0; i < numNoticias; i++) {
                Element tituloElem = elementosTitulo.get(i);
                Element fechaElem = elementosFecha.get(i);
                String titulo = tituloElem.text();
                String enlace = tituloElem.select("a").attr("abs:href");
                String fecha = fechaElem.text();
                noticias.add(new Noticia(titulo, fecha, enlace));
            }
        } catch (IOException e) {
            System.err.println("Error al acceder a la URL: " + e.getMessage());
        }
        return noticias;
    }
}
